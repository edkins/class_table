trait struct_str {
    fn field_names(self) -> list[str];
}

class Foo {
    a : str,
    b : str,
}

impl Foo :: struct_str {
    fn field_names(self) -> list[str] {
        ["a","b"]
    }
}

fn main(input: str) -> str {
    let x = Foo {
        a : "avalue",
        b : "bvalue",
    };
    let mut result = "";
    for name : x.field_names() {
        result += name;
    }
    assert(result == "ab");
    "ok"
}
