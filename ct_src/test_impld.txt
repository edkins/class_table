trait greeting {
    fn greet(self) -> str;
}

impl u32::greeting {
    fn greet(self) -> str {
        "hello"
    }
}

impl str::greeting {
    fn greet(self) -> str {
        "goodbye"
    }
}

fn main(input: str) -> str {
    let x = 42u32;
    let y = "string";
    assert(x.greet() + y.greet() == "hellogoodbye");
    "ok"
}
