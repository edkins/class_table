trait greeting {
    fn greet(self) -> str;
}

impl u32:a:greeting {
    fn greet(self) -> str {
        "hello"
    }
}

impl u32:b:greeting {
    fn greet(self) -> str {
        "goodbye"
    }
}

fn main(input: str) -> str {
    let x = 42u32;
    assert(a{x}.greet() + b{x}.greet() == "hellogoodbye");
    "ok"
}
